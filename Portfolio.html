// Simple FadeIn component using MUI's Fade

const FadeIn = ({ children, delay = 0 }) => {

    const [isIn, setIsIn] = useState(false);

    useEffect(() => {

        const timer = setTimeout(() => setIsIn(true), delay);

        return () => clearTimeout(timer);

    }, [delay]);


    // Ensure children is a valid React element before cloning

    if (!React.isValidElement(children)) {

        return <Fade in={isIn} timeout={800}>{children}</Fade>;

    }


    // Use Fade directly wrapping the children

    return <Fade in={isIn} timeout={800}>{children}</Fade>;

}
